自定义动态数组

构造方法

/**
 * 自定义动态数组
 * @author langao_q
 * @create 2020-05-19 15:34
 */
public class Array<E> {
    private E[] data;
    private int size;

    /**
     * 默认的构造函数，初始化长度为10
     */
    public Array(){
        this(10);
    }

    /**
     * 构造函数，传入的是数组的容量capacity构造Array
     * 此处数组类型不能指定为E泛型，泛型是从JDK1.5后才开始支持
     * @param capacity
     */
    public Array(int capacity){
        data = (E[]) new Object[capacity];
        size = 0;
    }

获取方法

    /**
     * 获取数组的大小
     * @return
     */
    public int getSize(){
        return size;
    }

    /**
     * 获取数组的最大容量
     * @return
     */
    public int getCapacity(){
        return data.length;
    }

    /**
     * 数组是否为空
     * @return
     */
    public boolean isEmpty(){
        return size==0;
    }
    
    /**
     * 获取指定位置元素
     * @param index
     * @return
     */
    public E get(int index){
        if(index < 0 || index > size){
            throw new IllegalArgumentException("add 添加元素失败 === index不合法！");
        }
        return data[index];
    }

    /**
     * 更新指定位置的元素
     * @param index
     * @param e
     */
    public void set(int index, E e){
        if(index < 0 || index > size){
            throw new IllegalArgumentException("add 添加元素失败 === index不合法！");
        }
        data[index] = e;
    }

    /**
     * 校验某个元素是否存在
     * @param e
     * @return
     */
    public boolean contains(E e){
        for (int i = 0; i < size; i++) {
            if(data[i].equals(e)){
                return true;
            }
        }
        return false;
    }

    /**
     * 查找指定元素的索引
     * @param e
     * @return
     */
    public int find(E e){
        for (int i = 0; i < size; i++) {
            if(data[i].equals(e)){
                return i;
            }
        }
        return -1;
    }

添加方法

    /**
     * 向数组头添加一个元素
     * @param e
     */
    public void addFirst(E e){
        add(0, e);
    }

    /**
     * 向数组末尾添加元素
     * @param e
     */
    public void addLast(E e){
        add(size, e);
    }

    /**
     * 向指定位置插入元素
     * @param index
     * @param e
     */
    public void add(int index, E e){
        if(index < 0 || index > size){
            throw new IllegalArgumentException("add 添加元素失败 === index不合法！");
        }
        if(size == data.length){
            resize(size * 2);
        }
        //把index位置后的每一个元素向后挪动一位
        for (int i = size -1; i >= index; i--) {
            data[i+1] = data[i];
        }

        data[index] = e;
        size ++;
    }

删除方法

    /**
     * 删除第一个元素
     * @return
     */
    public E removeFirst(){
        return remove(0);
    }

    /**
     * 删除最后一个元素
     * @return
     */
    public E removeLast(){
        return remove(size - 1);
    }

    /**
     * 删除指定元素
     * @param e
     */
    public boolean removeElement(E e){
        int index = find(e);
        if(index != -1){
            remove(index);
            return true;
        }
        return false;
    }

    /**
     * 删除指定位置的元素
     * @param index
     * @return
     */
    public E remove(int index){
        if(index < 0 || index > size){
            throw new IllegalArgumentException("add 添加元素失败 === index不合法！");
        }
        E result = data[index];
        //把index后的每一个元素 都向左移动一位
        for (int i = index+1; i < size ; i++) {
            data[i-1] = data[i];
        }
        size --;
        //loitering objects
        data[size] = null;
        if(size == data.length/2){
            resize(data.length/2);
        }
        return result;
    }

扩容方法

    /**
     * 数组扩容 每次size * 2
     * @param newCapacity
     */
    private void resize(int newCapacity){
        E[] newData = (E[]) new Object[newCapacity];
        for (int i = 0; i < size; i++) {
            newData[i] = data[i];
        }
        data = newData;
    }

    @Override
    public String toString() {
        return "Array{" +
                "data=" + Arrays.toString(data) +
                ", size=" + size +
                ", capacity=" + data.length +
                '}';
    }
